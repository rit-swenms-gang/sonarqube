    "*.jpg", "*.jpeg", "*.gif", "*.svg", "*.js", "*.js.map", "*.pdf", "/json/*", "*.woff2", "/static/*",
    "/robots.txt", "/favicon.ico", "/apple-touch-icon*", "/mstile*", "/vendor.LICENSE.txt");

  private StaticResources() {
    // only static
  }

  public static Collection<String> patterns() {
    return STATIC_RESOURCES;
  }
}
