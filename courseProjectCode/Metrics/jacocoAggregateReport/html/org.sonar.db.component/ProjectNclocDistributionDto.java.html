public class ProjectNclocDistributionDto {
  private String kee;
  private String name;
  private long ncloc;

  public String getKee() {
    return kee;
  }

  public ProjectNclocDistributionDto setKee(String kee) {
    this.kee = kee;
    return this;
  }

  public String getName() {
    return name;
  }

  public ProjectNclocDistributionDto setName(String name) {
    this.name = name;
    return this;
  }

  public long getNcloc() {
    return ncloc;
  }

  public ProjectNclocDistributionDto setNcloc(long ncloc) {
    this.ncloc = ncloc;
    return this;
  }
}
